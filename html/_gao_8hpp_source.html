27 class Failed_To_Create_Gaolette : public std::exception {
28 std::vector<std::string> msg_;

30 public:
31 explicit Failed_To_Create_Gaolette(int code) : code(code) {
32 msg_.emplace_back("Gaolette creation failed with error code: " + std::to_string(code));
33 if (code == -1) {
34 msg_.emplace_back("Unknown error occurred during Gaolette creation.");
35 } else if (code == 1) {
36 msg_.emplace_back("Invalid performance specification provided.");
37 } else if (code == 2) {
38 msg_.emplace_back("Insufficient system resources to create Gaolette.");
39 } else if (code == 3) {
40 msg_.emplace_back("Permission denied to create Gaolette.");
41 } else {
42 msg_.emplace_back("Unrecognized error code.");
43 }
44 }

47 return msg_.data()->c_str();
48 }

263 explicit Launch() {
264 int sv[2]; // socket pair
265 if (socketpair(AF_UNIX, SOCK_STREAM, 0, sv) == -1) {
266 throw std::runtime_error("socketpair failed");
267 }
268 posix_spawn_file_actions_init(&actions_);
269
270 // duplicate the child's socket to stdin/stdout
271 posix_spawn_file_actions_adddup2(&actions_, sv[1], STDIN_FILENO);
272 posix_spawn_file_actions_adddup2(&actions_, sv[1], STDOUT_FILENO);
273
274 status_ = posix_spawn(&pid_, get_gao_binary().c_str(),
275 &actions_, nullptr,
276 nullptr, environ);
277 if (status_ != 0) {
278 throw std::runtime_error("posix_spawn failed");
279 }
280
281 socket_ = sv[0];
282
283 // we can now close the child's end on the parent as it only needs it's end of the socket
284 close(sv[1]);
285 }
286
288 ~Launch() {
289 posix_spawn_file_actions_destroy(&actions_);
290 if (socket_ != -1) {
291 close(socket_);
292 }
293 if (pid_ > 0) {
294 if (waitpid(pid_, &status_, 0) > 0) {
295 if (WIFEXITED(status_)) {
296 std::cout << "Exited with status " << WEXITSTATUS(status_) << std::endl;
297 } else if (WIFSIGNALED(status_)) {
298 std::cout << "Killed by signal " << WTERMSIG(status_) << std::endl;
299 } else {
300 std::cout << "Exited with unknown status" << std::endl;
301 }
302 }

307 [[nodiscard]] std::string read_line() const {
308 constexpr size_t BUF_SIZE = 1024;
309 char* buf = new char[BUF_SIZE]; // caller must delete[]
310 size_t len = 0;
311
312 while (true) {
313 ssize_t nread = ::read(socket_, buf + len, BUF_SIZE - len - 1);
314 if (nread == -1) {
315 delete[] buf;
316 throw std::runtime_error("read failed");
317 }
318 if (nread == 0) { // EOF
319 break;
320 }
321 len += nread;
322
323 // stop if newline is seen
324 if (buf[len - 1] == '\n') {
325 buf[len - 1] = '\0';
326 std::string temp = buf;
327 delete[] buf;
328 return temp;
329 }
330
331 // safety: stop if buffer full (no newline)
332 if (len >= BUF_SIZE - 1) {
333 buf[len] = '\0';
334 std::string temp = buf;
335 delete[] buf;
336 return temp;
337 }
338 }
339
340 // no newline before EOF
341 buf[len] = '\0';
342 std::string temp = buf;
343 delete[] buf;
344 return temp;
345 }

354 inline Gaolette create_gaolette(Perf_Spec spec, const Launch& gao_p) {
355 std::string gao_instruction = "crt:";
356 gao_instruction.append(std::to_string(spec.size_) + ",");
357 gao_instruction.append(std::to_string(static_cast<int>(spec.memory_policy_)) + ",");
358 gao_instruction.append(std::to_string(spec.max_memory_usage_) + ",");
359 gao_instruction.append(std::to_string(spec.max_cpu_cores_));
360
361 gao_p.write_line(gao_instruction); // NOLINT
362
363 // we need to wait until read_line contains valid buffer space that we can read
364 const std::string response = gao_p.read_line();
365 if (response.empty() || response.substr(0, 2) == "-1") {
366 throw std::runtime_error("Gaolette creation failed");
367 }
368
369 // response comes in the form:
370 // OK:<gaolette_id_t>
371 // or
372 // ERR:<error_code>
373 if (response.substr(0, 3) == "ERR") {
374 int error_code = std::stoi(response.substr(4));
375 throw exceptions::Failed_To_Create_Gaolette(error_code);
376 }
377
378 if (response.substr(0, 2) == "OK") {
379 //parse until ,
380 gaolette_id_t id = std::stoi(response.substr(3));
381 return Gaolette{id, State::Operational, spec};
382 }
383
384 throw std::runtime_error("Invalid response from Gaolette creation");
385 }

401 inline void fetch_state(Gaolette& gaolette, Launch& gao_p) {
402 gao_p.write_line(std::string("get:state"));
403 std::string response = gao_p.read_line();
404 if (response.substr(0, 2) == "OK") {
405 int state_code = std::stoi(response.substr(3));
406 switch (state_code) {
407 case 0:
408 gaolette.state = State::Operational;
409 break;
410 case 1:
411 gaolette.state = State::ShutDown;
412 break;
413 case 2:
414 gaolette.state = State::Locked;
415 break;
416 case 3:
417 gaolette.state = State::Operating;
418 break;
419 case 4:
420 gaolette.state = State::Illformed;
421 break;
422 default:
423 throw std::runtime_error("Unknown state code received from Gaolette");
424 }
425 } else {
426 throw std::runtime_error("Failed to fetch Gaolette state");
427 }
428 }
